                                                                in org::csapi::fw::TpAuthDomain clientDomain,
                                                                in org::csapi::fw::TpAuthType authType,
                                                                in org::csapi::TpVersion frameworkVersion)

                                        interface IpAuthentication : org::csapi::IpInterface{
                                                org::csapi::IpInterface requestAccess(
                                                                in org::csapi::fw::TpAccessType accessType,

                                        }; // interface IpAuthentication

                                        interface IpClientAccess : org::csapi::IpInterface{
                                                void terminateAccess(
                                                                in org::csapi::TpString terminationText,
                                                                in org::csapi::fw::TpSigningAlgorithm signingAlgorithm,
                                                                in org::csapi::TpOctetSet digitalSignature)

                                        interface IpAccess : org::csapi::IpInterface{
                                                org::csapi::IpInterface obtainInterface(
                                                                in org::csapi::fw::TpInterfaceName interfaceName)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED, org::csapi::P_INVALID_INTERFACE_NAME);
                                                org::csapi::IpInterface obtainInterfaceWithCallback(
                                                                in org::csapi::fw::TpInterfaceName interfaceName,
                                                                in org::csapi::IpInterface clientInterface)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED, org::csapi::P_INVALID_INTERFACE_NAME, org::csapi::P_INVALID_INTERFACE_TYPE);
                                                void endAccess(
                                                                in org::csapi::fw::TpEndAccessProperties endAccessProperties)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED, org::csapi::fw::P_INVALID_PROPERTY);
                                                org::csapi::fw::TpInterfaceNameList listInterfaces()
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED);
                                                void releaseInterface(
                                                                in org::csapi::fw::TpInterfaceName interfaceName)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED, org::csapi::P_INVALID_INTERFACE_NAME);
                                                org::csapi::fw::TpSigningAlgorithm selectSigningAlgorithm(
                                                                in org::csapi::fw::TpSigningAlgorithmCapabilityList signingAlgorithmCaps)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_ACCESS_DENIED, org::csapi::fw::P_NO_ACCEPTABLE_SIGNING_ALGORITHM);
                                                void terminateAccess(
                                                                in org::csapi::TpString terminationText,
                                                                in org::csapi::TpOctetSet digitalSignature)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_INVALID_SIGNATURE);
                                                void relinquishInterface(
                                                                in org::csapi::fw::TpInterfaceName interfaceName,
                                                                in org::csapi::TpString terminationText,
                                                                in org::csapi::TpOctetSet digitalSignature)
                                                                raises (org::csapi::TpCommonExceptions, org::csapi::fw::P_INVALID_SIGNATURE, org::csapi::P_INVALID_INTERFACE_NAME);
                                        }; // interface IpAccess

                                                                in org::csapi::TpOctetSet challenge)

                                }; // module trust_and_security

                        }; // module fw_access
